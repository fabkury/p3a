# Flash command for ESP32-P4
# Run from this directory:
python -m esptool --chip esp32p4 -b 460800 --before default_reset --after hard_reset write_flash --force @flash_args

# Or with explicit port (replace COM5 with your port):
python -m esptool --chip esp32p4 -p COM5 -b 460800 --before default_reset --after hard_reset write_flash --force @flash_args
